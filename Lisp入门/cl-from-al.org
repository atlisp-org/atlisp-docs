#+title: 从autolisp到commonlisp
* 局部变量
#+begin_src lisp
  (defun foobar ()
    (let ((a 5)
	  (b 6))
      (+ a b)))
#+end_src
#+begin_src autolisp
  (defun foobar (/ a b)
    (setq a 5 b 6)
    (+ a b))
#+end_src
* let 与 let*
let* 可以使用前面定义的局部变量,let 不可以。
#+begin_src lisp
  (let ((a 5)
        (b 6))
    (+ a b))

  (let* ((a 5)
         (b (1+ a)))
    (+ a b))
#+end_src
